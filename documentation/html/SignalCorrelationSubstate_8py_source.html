137  ## @fun #computeSignalDelay computes the delay between the #trueSignal and
138  # measurements based on a correlation vector
139  #
140  # @details The #computeSignalDelay function is a rudimentary function
141  # which takes an estimate of the correlation vector and uses it to
142  # estimate the location of the peak. It functions by finding the tap with
143  # the maximum value, and then fitting a quadratic to the points
144  # surrounding the maximum value tap. The number of points to which the
145  # quadratic is fitted is determined by the value of #peakFitPoints; the
146  # number of points is equal to \f$2 * n + 1\f$ where \f$n = \f$
147  # #peakFitPoints.
148  #
149  # The delay estimate that is returned is in units of #dT. So, a returned
150  # value of 2 would imply that the peak is located at 2, and therefore the
151  # delay corresponding to the correlation vector is 2 #dT.
152  #
153  # The returned delay may not include previously accumulated #signalDelay
154  # between the signal and the measurements. See the #storeStateVector
155  # function for more information on how the #signalDelay is stored and
156  # accumulated delay is accounted for.
157  #
158  # @param self The object pointer
159  # @param c The correlation vector
160  # @param P the correlation vector covariance matrix
161  #
162  # @return The estimate of the delay
163  def computeSignalDelay(
164  self,
165  c,
166  P
167  ):
168 
169  # First estimate of peak location is the location of the max value
170  peakLocation = np.argmax(c)
171 
172  # Next, we "roll" the correlation vector so that the values being
173  # fitted quadratically are the first 2 * peakFitPoints + 1 values
174  rollFactor = self.peakFitPoints - peakLocation
175 
176  if rollFactor != 0:
177  slicedC = np.roll(
178  c,
179  rollFactor
180  )
181  slicedP = (
182  np.roll(
183  np.roll(P,
184  rollFactor,
185  axis=0),
186  rollFactor,
187  axis=1
188  )
189  )
190  else:
191  slicedC = c
192  slicedP = P
193 
194  # Extract the portion of the correlation vector used for fitting
195  # quadratic
196  mySlice = slice(0, self.peakFitPoints * 2 + 1)
197 
198  slicedC = slicedC[mySlice]
199  slicedP = slicedP[mySlice, mySlice]
200 
201  # np.polyfit assumes that the weights will be the inverse standard
202  # deviation
203  weightVector = np.diag(slicedP)
204  weightVector = 1 / np.sqrt(weightVector)
205 
206  # xVec is the vector of "x" values corresponding the "y" values to
207  # which the quadratic is being fit.
208  xVec = np.linspace(0, self.peakFitPoints * 2, (self.peakFitPoints * 2) + 1)
209  xVec = xVec - rollFactor
210 
211  # Get the quadratic function that fits the peak and surrounding values,
212  # and use it to estimate the location of the max
213  quadraticVec = np.polyfit(xVec, slicedC, 2, w=weightVector)
214  delay = (-quadraticVec[1] / (2 * quadraticVec[0]))
215 
216  return delay
