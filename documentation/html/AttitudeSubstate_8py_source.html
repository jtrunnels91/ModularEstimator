18 ## @class AttitudeState6DOF
19 # @brief Estimates the attitude of a vehicle in three dimensions, along with
20 # three gyro bias states.
21 #
22 # @details
23 # This class contains a six-state attitude estimator: three attitude states
24 # and three gyro bias states.
25 #
26 # This class can function as a stand-alone class, or it can function as a
27 # "SubState" of the State.ModularFilter class. The functions required for use
28 # as a SubState are defined first after __init__, then functions specific to
29 # this class are defined next.
30 #
31 # The state uses quaternions to store attitude, which avoids issues of gimbal
32 # lock and increases numerical stability over other approaches, such as Euler
33 # angles. The quaternion itself is not treated as a part of the state vector.
34 # Rather, the state vector includes three attitude "error states," which are
35 # updated at each measurement, then used to correct the attitude quaternion.
36 # After each correction, the error states are set back to zero.
37 #
38 # The algorithms used for the state update mostly come from the book
39 # "Fundamentals of Spacecraft Attitude Determination and Control" (FSADC) by
40 # Markley and Crassidis. Chapter, section and page numbers will be referenced
41 # where appropriate.
42 class AttitudeState6DOF(SubState):

367  ## @fun quaternionTimeUpdateMatrix produces a time-update matrix for the
368  # attitude quaternion
369  #
370  # @details This function produces a 4x4 matrix which, when multiplied by
371  # an attitude quaternion, rotates the quaternion by an amount
372  # corresponding to the angular velocity and time ellapsed. The attitude
373  # quaternion is updated as follows:
374  #
375  # \f[
376  # \attVec[est=True,aPriori=True, t=k+1] \approx
377  # \bar{\Theta}(\omegaVec[est=True,aPriori=True, t=k], \Delta T)
378  # \attVec[est=True, aPriori=False, t=k]
379  # \f]
380  #
381  # where
382  #
383  # \f[
384  # \bar{\Theta}( \omegaVec[est=True,aPriori=True, t=k], \Delta T ) =
385  # \begin{bmatrix}
386  # \textrm{cos} \left(\frac{1}{2}
387  # ||\omegaVec[est=True,aPriori=True, t=k]|| \Delta t \right) I_3 -
388  # \left[\boldsymbol{\hat{\Psi}}_k^+ \times \right]
389  # & \left[\boldsymbol{\hat{\Psi}}_k^+ \times \right] \\
390  # - \left[\boldsymbol{\hat{\Psi}}_k^+ \times \right] &
391  # \textrm{cos} \left(\frac{1}{2} ||\mathbf{\hat{\omega}}_k^+|| \Delta t \right)
392  # \end{bmatrix}
393  # \f]
394  #
395  # and
396  #
397  # \f[
398  # \left[\boldsymbol{\hat{\Psi}}_k^+ \times \right] =
399  # \frac{
400  # \textrm{sin}\left(\frac{1}{2} || \omegaVec[est=True,aPriori=True, t=k] || \Delta t \right)
401  # \omegaVec[est=True,aPriori=True, t=k]
402  # }{
403  # || \omegaVec[est=True,aPriori=True, t=k] ||
404  # }
405  # \f]
406  #
407  # The matrix returned by this function is
408  # \f$\bar{\Theta}(\omegaVec[est=True,aPriori=True, t=k], \Delta T)\f$.
409  #
410  # See Fundamentals of Spacecraft Attitude Determination and Control,
411  # Section 6.2.2, page 251, equation 6.60 for more details.
412  #
413  # @param self The object pointer
414  # @param myOmega The angular velocity estimate used to update the attitude
415  # quaternion
416  # @param deltaT The amount of time elapsed for the time-update, used for
417  # numerical integration of kinematics equation.
418  #
419  # @return The quaternion time-update matrix
420  # \f$\bar{\Theta}(\omegaVec[est=True,aPriori=True, t=k], \Delta T)\f$
421  def quaternionTimeUpdateMatrix(
422  self,
423  myOmega,
424  deltaT
425  ):
426  omegaNorm = norm(myOmega)
427  cosineTerm = cos(0.5 * omegaNorm * deltaT)
428  if abs(omegaNorm) < 1e-100:
429  psiK = np.zeros(3)
430  else:
431  psiK = (sin(0.5 * omegaNorm * deltaT) / omegaNorm) * myOmega
432  theta = np.zeros([4,4])
433 
434  theta[1:4, 1:4] = cosineTerm*np.eye(3) - self.skewSymmetric(psiK)
435 
436  theta[0, 1:4] = -psiK
437  theta[1:4, 0] = psiK
438  theta[0, 0] = cosineTerm
439  return theta

441  ## @fun errorStateTimeUpdateMatrix produces a time-update matrix for the
442  # attitude error state
443  #
444  # @details
445  # This function the discrete-time error-state transition matrix. This is
446  # the matrix which propagates the attitude error state covariance and gyro
447  # bias covariance forward in time based on time ellapsed and angular
448  # velocity estimate.
449  #
450  # The error-state transition matrix is defined as follows:
451  #
452  # \f[
453  # \boldsymbol{\Phi} = \begin{bmatrix}
454  # \boldsymbol{\Phi}_{11} & \boldsymbol{\Phi}_{12} \\
455  # \boldsymbol{\Phi}_{21} & \boldsymbol{\Phi}_{22} \\
456  # \end{bmatrix}
457  # \f]
458  #
459  # where
460  #
461  # \f[
462  # \boldsymbol{\Phi}_{11} = \eye[3] -
463  # \left[\omegaVec[est=True,aPriori=True, t=k] \times \right]
464  # \frac
465  # {\textrm{sin}(||\omegaVec[est=True,aPriori=True, t=k]|| \Delta t)}
466  # {||\omegaVec[est=True,aPriori=True, t=k]||} +
467  # \left[\omegaVec[est=True,aPriori=True, t=k] \times \right]^2
468  # \frac
469  # {1 - \textrm{cos}(1 - ||\omegaVec[est=True,aPriori=True, t=k]|| \Delta t)}
470  # {||\omegaVec[est=True,aPriori=True, t=k]||^2}
471  # \f]
472  #
473  # \f[
474  # \boldsymbol{\Phi}_{12} =
475  # \left[\omegaVec[est=True,aPriori=True, t=k] \times \right]
476  # \frac
477  # {1 - \textrm{cos}(1 - ||\omegaVec[est=True,aPriori=True, t=k]|| \Delta t)}
478  # {||\omegaVec[est=True,aPriori=True, t=k]||^2} -
479  # \eye[3]\Delta t -
480  # \left[\omegaVec[est=True,aPriori=True, t=k] \times \right]^2
481  # \frac
482  # {||\omegaVec[est=True,aPriori=True, t=k]|| \Delta t -
483  # \textrm{sin}(||\omegaVec[est=True,aPriori=True, t=k]|| \Delta t)}
484  # {||\omegaVec[est=True,aPriori=True, t=k]||^3}
485  # \f]
486  #
487  # \f[
488  # \boldsymbol{\Phi}_{21} = \mathbf{0}_{3 \times 3}
489  # \f]
490  #
491  # \f[
492  # \boldsymbol{\Phi}_{22} = \eye[3]
493  # \f]
494  #
495  # See Fundamentals of Spacecraft Attitude Determination and Control,
496  # Section 6.2.4, page 258, equation 6.83 for more details and derivation.
497  #
498  # @param self The object pointer
499  # @param myOmega The angular velocity estimate used to update the attitude
500  # quaternion
501  # @param deltaT The amount of time elapsed for the time-update, used for
502  # numerical integration of kinematics equation.
503  #
504  # @return Returns the error-state time update matrix, \f$\boldsymbol{\Phi}\f$
505  def errorStateTimeUpdateMatrix(
506  self,
507  myOmega,
508  deltaT
509  ):
510  omegaNorm = norm(myOmega)
511  omegaNormSquare = square(omegaNorm)
512  omegaNormDT = omegaNorm * deltaT
513  omegaSkew = self.skewSymmetric(myOmega)
514  omegaSkewSquare = power(omegaSkew, 2)
515 
516  # Check to see if omega is very close to zero. If it is, avoid error
517  # dividing by zero.
518  if omegaNorm > 1e-100:
519  phi11 = (
520  np.eye(3) -
521  (omegaSkew * sin(omegaNormDT)/omegaNorm) +
522  (omegaSkewSquare * (1 - cos(omegaNormDT))/omegaNormSquare)
523  )
524  phi12 = (
525  (omegaSkew * (1 - cos(omegaNormDT))/omegaNormSquare) -
526  (np.eye(3) * deltaT) -
527  (omegaSkewSquare * (omegaNormDT - sin(omegaNormDT))/power(omegaNorm,3))
528  )
529  else:
530  phi11 = (
531  np.eye(3)
532  )
533 
534  phi12 = (
535  (-np.eye(3) * deltaT)
536  )
537 
538  phi = np.vstack([
539  np.concatenate([phi11, phi12], axis=1),
540  np.concatenate([np.zeros([3,3]), np.eye(3)], axis=1)
541  ])
542 
543  return(phi)

545  ## @fun processNoiseMatrix generates a the process noise matrix
546  #
547  # @details
548  # This function generates the process noise matrix for time update of
549  # attitude error covariance and gyro bias covariance. The process noise
550  # matrix is a function propagation time, angular velocity noise, and gyro
551  # bias noise. It is defined as follows:
552  #
553  # \f[
554  # \mathbf{Q} = \begin{bmatrix}
555  # \left(\sigma_v^2 \Delta t + \frac{1}{3}\sigma_u^2 \Delta t^3\right) \eye[3] &
556  # -\left( \frac{1}{2} \sigma_u^2 \Delta t^2 \right) \eye[3] \\
557  # -\left( \frac{1}{2} \sigma_u^2 \Delta t^2 \right) \eye[3] &
558  # \left( \sigma_u^2 \Delta t \right) \eye[3]
559  # \end{bmatrix}
560  # \f]
561  #
562  # where \f$\sigma_v^2\f$ is the angular velocity noise (i.e. gyro
563  # measurement noise) and \f$ \sigma_u^2 \f$ is the gyro bias process noise.
564  #
565  # See Fundamentals of Spacecraft Attitude Determination and Control,
566  # Section 6.2.4, page 260, equation 6.93 for derivation and more details.
567  #
568  # @param self The object pointer
569  # @param deltaT The amount of time corresponding to the time update
570  # @param omegaVar The variance of the angular velocity (gyro) measurement
571  # @param biasVar The variance of the gias bias process noise (indicates
572  # how much the gyro bias changes over time)
573  # @return Returns the comibined 6x6 process noise matrix
574  def processNoiseMatrix(
575  self,
576  deltaT,
577  omegaVar,
578  biasVar
579  ):
580  deltaTSquared = power(deltaT, 2)
581  Q11 = (
582  (omegaVar * deltaT) +
583  (biasVar * power(deltaT, 3))/3
584  ) * np.eye(3)
585 
586  Q12 = (
587  -(biasVar * deltaTSquared)/2
588  ) * np.eye(3)
589 
590  Q21 = Q12
591 
592  Q22 = (biasVar * deltaT) * np.eye(3)
593 
594  Q = np.vstack([
595  np.concatenate([Q11, Q12], axis=1),
596  np.concatenate([Q21, Q22], axis=1)
597  ])
598 
599  return(Q)

601  ## @fun RaDecMeasMatrices generates measurement matrices for a angle
602  # measurement of a point source.
603  #
604  # @details
605  # This function generates the set of measurement matrices
606  # \f$ \mathbf{H} \f$, \f$ \mathbf{dY} \f$, and \f$ \mathbf{R} \f$
607  # corresponding to an inferred unit vector measurement from a set of two
608  # angle measurments (local right ascension and declination of a point
609  # source).
610  #
611  # The measurement matrices are a function of the measurement itself, the
612  # source from which the measurement originated, and the current estimate
613  # of attitude, #qHat. They are defined as follows:
614  #
615  # \f[
616  # \mathbf{H}_k[\sv[aPriori=True,timeIndex=k]] =
617  # A(\attVec[est=True,aPriori=True,t=k])
618  # \unitVec[signalSource=S, frame=nav] \times
619  # \f]
620  #
621  # \f[
622  # \mathbf{dY} =
623  # \measurementVec[S](\RADEC) -
624  # A(\attVec[est=True,aPriori=True,t=k])
625  # \unitVec[signalSource=S, frame=nav]
626  # \f]
627  #
628  # \f[
629  # \mathbf{R} = \eye[3] \sigma^2_{\RADEC}
630  # \f]
631  #
632  # The measured unit vector is a unit vector computed from the measured
633  # angles, using #sidUnitVec.
634  #
635  # See Fundamentals of Spacecraft Attitude Determination and Control,
636  # Section 6.2.4, page 257, Table 6.3 for more details and derivation.
637  #
638  # @note Currently, the measurement noise matrix is an identity matrix
639  # mutiplied by the angle measurement error. This is my interpretation of
640  # Section 6.2.3 in FSADC. However, if we derive the measurement noise
641  # matrix using the usual EKF method, we get a different result which
642  # results in an unstable estimator. I'm not convinced I understand why

726  ## @fun eulerAngles computes the Euler angles (roll, pitch and yaw) based
727  # on the current attitude.
728  #
729  # @details This function computes the Euler angles (or, technically the
730  # "Tait-Bryan angles"), i.e. roll, pitch and yaw from the current attitude
731  # quaternion #qHat.
732  #
733  # @note See Wikipedia's article on the
734  # <a href="https://en.wikipedia.org/wiki/Euler_angles">
735  # equatorial coordinate system</a> for more details.
736  #
737  # @param self The object pointer
738  #
739  # @returns A list containing the three angles
740  def eulerAngles(self):
741  q = self.qHat
742  phi = arctan2(2*(q[0]*q[1] + q[2]*q[3]), 1 - 2*(square(q[1]) + square(q[2])))
743  theta = arcsin(2 * ((q[0] * q[2]) - (q[3] * q[1])))
744  psi = arctan2(2 * (q[0] * q[3] + q[1]*q[2]), 1 - 2*(square(q[2]) + square(q[3])))
745 
746  return [phi, theta, psi]
