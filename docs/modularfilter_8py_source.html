30 # @details
31 # This class is designed to facilitate a variety of estimation algorithms in a
32 # modular way, i.e. in a way that allows maximum amount of flexibility with
33 # minimum amount of code duplication.
34 #
35 # The idea behind this class is that all of the functions which are "generic"
36 # to an estimation algorithm can be written just once, as a part of this class.
37 # Then the code contained here can be implemented on a variety of different
38 # estimation problems without having to duplicate code.
39 #
40 # The basic estimation model implemented here works as follows. The overall
41 # state, i.e. all of the quantities to be estimated, are represented by
42 # SubStates.SubState objects. Measurements of these states are represented by
43 # Signals.SignalSource objects. The ModularFilter class is responsible for doing time-updates and measurement-updates on these states.
44 # These objects are responsible for doing all
45 # of the things that are particular to those states, and those signals,
46 # respectively. For instance, generation of time-update and measurement
47 # update matrices is handled by the SubStates.SubState objects.

791  def computeUpdatedStateandCovariance(
792  self,
793  xMinus,
794  PMinus,
795  dY,
796  H,
797  R
798  ):
799  if PMinus.form == 'covariance':
800  # Standard Kalman Filter
801  S = H.dot(PMinus.value).dot(H.transpose()) + R
802 
803  # Could inversion of S be introducting instability?
804  K = PMinus.value.dot(H.transpose()).dot(np.linalg.inv(S))
805 
806  IminusKH = np.eye(self.totalDimension) - K.dot(H)
807 
808  xPlus = xMinus + K.dot(dY)
809  PPlus = (
810  IminusKH.dot(PMinus.value).dot(IminusKH.transpose()) +
811  K.dot(R).dot(K.transpose())
812  )
813  PPlus = covarianceContainer(PPlus, 'covariance')
814  elif PMinus.form == 'cholesky':
815 
816  W = PMinus.value
817  Z = W.transpose().dot(H.transpose())
818 
819  # Compute U. Instead of using cholesky decomposition however, use
820  # LDL (since R + Z^TZ can be semidefinite)
821  # U = np.linalg.cholesky(R + Z.transpose().dot(Z))
822 
823  R_plus_ZTZ = R + Z.transpose().dot(Z)
824  myLDL = ldl(R_plus_ZTZ)
825  U = myLDL[0].dot(np.sqrt(myLDL[1]))
826  # Compute V
827  # V = np.linalg.cholesky(R)
828  myLDL = ldl(R)
829  V = myLDL[0].dot(np.sqrt(myLDL[1]))
830  UInv = np.linalg.inv(U)
831 
832  WPlus = W.dot(
833  np.eye(self.totalDimension) -
834  Z.dot(UInv.transpose()).dot(np.linalg.inv(U + V)).dot(Z.transpose())
835  )
836  PPlus = covarianceContainer(WPlus, PMinus.form)
837  xPlus = xMinus + W.dot(Z).dot(UInv.transpose()).dot(UInv).dot(dY)
838  if np.any([isnan(stateVal) for stateVal in xPlus]):
839  print(xPlus)
840  raise ValueError(
841  'Computed a NaN updated state vector'
842  )
843  return (xPlus, PPlus)
