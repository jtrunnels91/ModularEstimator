388  ## @fun #getSignal is responsible for returning the photon flux from the
389  # pulsar at a given time
390  #
391  # @details The #getSignal method is the method which is responsible for
392  # returning the current value of flux from the signal source at a given
393  # time. If there is uncertainty in the time, then the expected value of
394  # the signal is returned. Uncertainty in time is indicated by the
395  # optional tVar argument.
396  #
397  # In every case, the method calls the #getPhase method to determine the
398  # current signal source phase. If no uncertainty in time is passed, then
399  # the #getPulseFromPhase method is called to lookup/interpolate the
400  # current flux based on the phase and the signal #profile.
401  #
402  # If a value is passed for tVar, then the process is more complicated.
403  # Rather than simply look up the signal value from the phase, the function
404  # returns an approximation of the expected value of the signal, given the
405  # mean and variance of time.
406  #
407  # Normally, the expected value would be computed by integrating the
408  # product of the true distribution of time (probably a normal distribution)
409  # with the flux as a function of time. However, due to the non-analytical
410  # nature of the flux function, the direct computation of this integral is
411  # intractable. To overcome this limitation, the time distributuion is
412  # approximated as a moment-matched uniform distribution. Using this
413  # approximation, the approximate integral may be directly computed simply
414  # by looking up the start and end values of #singlePeriodIntegral, and
415  # dividing by the appropriate scaling factor.
416  #
417  # @param self The object pointer
418  # @param observatoryTime The time for which to compute the signal
419  # @param tVar (optional) The variance of the time estimate
420  #
421  # @return The signal at the requested time
422  def getSignal(
423  self,
424  observatoryTime,
425  tVar=None
426  ):
427  # Get the phase corresponding to the current time
428  phase = self.getPhase(observatoryTime)
429 
430  # If a value was received for the tVar, then we compute the expected
431  # value of flux
432  if tVar is not None:
433 
434  # Get standard deviation of t
435  tSigma = np.sqrt(tVar)
436 
437  # Convert the time standard deviation to phase standard deviation
438  phaseSigma = tSigma/self.pulsarPeriod
439 
440  # Check to see if the phase std is bigger than the std
441  # corresponding to a uniform distribution with support = 1. If so,
442  # this indicates that we effectively have no meaningful knowledge
443  # of phase, and can just return the average flux.
444  if phaseSigma > np.sqrt(1/12):
445  signal = self.avgPhotonFlux * self.pulsedFraction
446  else:
447  phaseFraction = np.mod(phase, 1.0)
448  upperSigma = phaseFraction + (np.sqrt(12) * phaseSigma / 2)
449  lowerSigma = phaseFraction - (np.sqrt(12) * phaseSigma / 2)
450  upperSigmaOffset = 0
451  lowerSigmaOffset = 0
452  if upperSigma > 1:
453  upperSigma = upperSigma - 1
454  upperSigmaOffset = self.singlePeriodIntegral[-1]
455  if lowerSigma < 0:
456  lowerSigma = lowerSigma + 1
457  lowerSigmaOffset = self.singlePeriodIntegral[-1]
458  signal = (
459  upperSigmaOffset +
460  np.interp(
461  upperSigma,
462  self.profileIndex,
463  self.singlePeriodIntegral
464  ) -
465  np.interp(
466  lowerSigma,
467  self.profileIndex,
468  self.singlePeriodIntegral)
469  + lowerSigmaOffset
470  )
471  signal = (
472  signal *
473  self.peakAmplitude /
474  (np.sqrt(12) * tSigma)
475  )
476  else:
477  signal = self.getPulseFromPhase(phase)
478 
479  if self.backgroundCountRate is not None:
480  signal = signal + self.backgroundCountRate
481 
482  return(signal)
