326  def computeSinglePeriodIntegral(
327  self
328  ):
329  # Compute the integral as a function of time of the pulsar flux. This
330  # will be used later to compute expected value of flux in the case
331  # where time is uncertain.
332  self.singlePeriodIntegral = np.zeros(len(self.profile))
333 
334  singlePeriodTimeArray = np.linspace(
335  0, self.pulsarPeriod, len(self.profile)

339 
340  for i in range(len(self.profile)):
341

343  self.profile[0:i + 1],
344  singlePeriodTimeArray[0:i + 1],
345  axis=0
346  )
347 
348  # Store the total flux integral over one period
349  fluxIntegral = self.singlePeriodIntegral[-1]
350 
351  # Compute the scaling factor based on the average photon flux (if the
352  # average flux was given)
353  if self.avgPhotonFlux is not None:
354 
355  # Compute number of photons for a given pulsar period
356  self.photonsPerPeriod = (
357  self.avgPhotonFlux * self.pulsarPeriod
358  )
359 
360  # Scale the peak amplitude so that a single integrated pulse will
361  # result in the expected number of photons
362  self.peakAmplitude = self.photonsPerPeriod / fluxIntegral
363

365  # accordingly and compute a background rate to account for the

415  ## @fun #getSignal is responsible for returning the photon flux from the
416  # pulsar at a given time
417  #
418  # @details The #getSignal method is the method which is responsible for
419  # returning the current value of flux from the signal source at a given
420  # time. If there is uncertainty in the time, then the expected value of
421  # the signal is returned. Uncertainty in time is indicated by the
422  # optional tVar argument.
423  #
424  # In every case, the method calls the #getPhase method to determine the
425  # current signal source phase. If no uncertainty in time is passed, then
426  # the #getPulseFromPhase method is called to lookup/interpolate the
427  # current flux based on the phase and the signal #profile.
428  #
429  # If a value is passed for tVar, then the process is more complicated.
430  # Rather than simply look up the signal value from the phase, the function
431  # returns an approximation of the expected value of the signal, given the
432  # mean and variance of time.
433  #
434  # Normally, the expected value would be computed by integrating the
435  # product of the true distribution of time (probably a normal distribution)
436  # with the flux as a function of time. However, due to the non-analytical
437  # nature of the flux function, the direct computation of this integral is
438  # intractable. To overcome this limitation, the time distributuion is
439  # approximated as a moment-matched uniform distribution. Using this
440  # approximation, the approximate integral may be directly computed simply
441  # by looking up the start and end values of #singlePeriodIntegral, and
442  # dividing by the appropriate scaling factor.
443  #
444  # @param self The object pointer
445  # @param observatoryTime The time for which to compute the signal
446  # @param tVar (optional) The variance of the time estimate
447  #
448  # @return The signal at the requested time
449  def getSignal(
450  self,
451  observatoryTime,
452  tVar=None,
453  state=None
454  ):
455  if state is not None:
456  if 'signalDelay' in state:
457  delay = state['signalDelay']
458  if 'delayVar' in state:
459  delayVar = state['delayVar']
460  else:
461  delayVar = 0
462  observatoryTime = observatoryTime + delay
463  tVar = tVar + delayVar
464  # Get the phase corresponding to the current time
465  phase = self.getPhase(observatoryTime)
466 
467  # If a value was received for the tVar, then we compute the expected
468  # value of flux
469  if tVar is not None:
470 
471  # Get standard deviation of t
472  tSigma = np.sqrt(tVar)
473 
474  # Convert the time standard deviation to phase standard deviation
475  phaseSigma = tSigma/self.pulsarPeriod
476 
477  # Check to see if the phase std is bigger than the std
478  # corresponding to a uniform distribution with support = 1. If so,
479  # this indicates that we effectively have no meaningful knowledge
480  # of phase, and can just return the average flux.
481  if phaseSigma > np.sqrt(1/12):
482  signal = self.avgPhotonFlux * self.pulsedFraction
483  elif phaseSigma < 1/(100 * self.profileLen):
484  signal = self.getPulseFromPhase(phase)
485 
486  else:
487  phaseFraction = np.mod(phase, 1.0)
488  upperSigma = phaseFraction + (np.sqrt(12) * phaseSigma / 2)
489  lowerSigma = phaseFraction - (np.sqrt(12) * phaseSigma / 2)
490  upperSigmaOffset = 0
491  lowerSigmaOffset = 0
492  if upperSigma > 1:
493  upperSigma = upperSigma - 1
494  upperSigmaOffset = self.singlePeriodIntegral[-1]
495  if lowerSigma < 0:
496  lowerSigma = lowerSigma + 1
497  lowerSigmaOffset = self.singlePeriodIntegral[-1]
498  signal = (
499  upperSigmaOffset +
500  np.interp(
501  upperSigma,
502  self.profileIndex,
503  self.singlePeriodIntegral
504  ) -
505  np.interp(
506  lowerSigma,
507  self.profileIndex,
508  self.singlePeriodIntegral)
509  + lowerSigmaOffset
510  )
511  signal = (
512  signal /
513  (np.sqrt(12) * tSigma)
514  )
515  signal = signal * self.scaleFactor
516 
517  else:
518  signal = self.getPulseFromPhase(phase)
519 
520  if self.backgroundCountRate is not None:
521  signal = signal + self.backgroundCountRate
522 
523  return(signal)
