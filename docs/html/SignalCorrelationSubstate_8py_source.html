16 ## @class CorrelationFilter
17 # @brief CorrelationFilter estimates the correlation vector and delay between
18 # a signal and a time-delayed measurement of that signal
19 #
20 # @details
21 # This class contains an estimator which estimates the correlation vector
22 # between a signal (the #__trueSignal__) and measurements of that signal. This
23 # correlation vector is then used to estimate the delay between the
24 # #__trueSignal__ and the measurements of that signal.
25 #
26 # The estimator in this class currently assumes that the signal source is
27 # "distant," or infinitely far away. This implies that the unit vector to the
28 # signal source is perfectly known, and not changing. A later implementation
29 # could include the option of having a non-distant signal source, in which the
30 # unit vector is changing as a function of position and therefore uncertain.
31 #
32 # @note This class is essentially an implementation of the estimator presented in
33 # <a href="https://doi.org/10.2514/1.G002650">
34 # Recursive Range Estimation Using Astrophysical Signals of Opportunity</a>,
35 # J. Runnels, D. Gebre, Journal of Guidance, Control and Dynamics, 2017. Some
36 # equations from the paper are included in the class documentation for
37 # reference. A more detailed discussion and derivation of the estimator can
38 # be found in the journal article..

310  ## @fun #buildTimeUpdateMatrices constructs the correlation vector time
311  # update matrices
312  #
313  # @details The #buildTimeUpdateMatrices method constructs the matrices required to perform the time update of the correlation vector sub-state.
314  #
315  # The time update matrices are a function of the estimated spacecraft velocity (\f$\mathbf{v}\f$), velocity variance (\f$\mathbf{P}_{\mathbf{v}}\f$), and the elapsed time over which the time update occurs (\f$\Delta T\f$). The matrices are constructed as follows:
316  #
317  # \f[
318  # \mathbf{F}_{j \to k} = \begin{bmatrix}
319  # \textrm{sinc}(\hat{\delta}) & \hdots & \textrm{sinc}(\hat{\delta} + N - 1) \\
320  # \vdots & \ddots & \vdots \\
321  # \textrm{sinc}(\hat{\delta} - N + 1) & \hdots & \textrm{sinc}(\hat{\delta})
322  # \end{bmatrix}
323  # \f]
324  #
325  # \f[
326  # \mathbf{L}_{j} = \begin{bmatrix}
327  # \frac{\textrm{cos}}{(\hat{\delta})} - \frac{\textrm{sin}}{(\hat{\delta}^2)} & \hdots \\
328  # \vdots & \ddots \\
329  # \end{bmatrix} \sv[timeIndex = k]
330  # \f]
331  #
332  # \f[
333  # Q_{\delta} = \left(\frac{(k-j)}{c}\right)^2
334  # {\LOSVec[S]}^T \mathbf{P}_{\mathbf{v}} \LOSVec[S]
335  # \f]
336  #
337  # where
338  #
339  # \f[
340  # \hat{\delta}_{j \to k} = \frac{\mathbf{v} \LOSVec[S] \Delta T}{c T}
341  # \f]
342  #
343  # @param self The object pointer
344  # @param deltaT The amount of time over which the time update is occuring
345  # @param dynamics A dictionary containing the relevant dynamics for the
346  # time update
347  # @param h The current correlation vector
348  #
349  # @returns A dictionary containing the matrices \f$\mathbf{F}\f$,
350  # \f$\mathbf{L}\f$, and the scalar \f$Q\f
351  def buildTimeUpdateMatrices(
352  self,
353  deltaT,

358

365  (velocity.dot(self.__unitVecToSignal__) * indexDiff) /
366  self.speedOfLight()
367  )
368 
369  Q = (
370  self.__unitVecToSignal__.dot(
371  vVar
372  ).dot(self.__unitVecToSignal__) *
373  np.square(indexDiff / self.speedOfLight())

377  F = np.zeros([self.__filterOrder__, self.__filterOrder__])

381  halfLength = np.ceil(self.__filterOrder__ / 2)
382 
383  if np.mod(self.__filterOrder__, 2) == 0:

385  np.linspace(
386  1 - halfLength,

388  self.__filterOrder__

482  ## @fun #computeSignalDelay computes the delay between the #__trueSignal__ and
483  # measurements based on a correlation vector
484  #
485  # @details The #computeSignalDelay function is a rudimentary function
486  # which takes an estimate of the correlation vector and uses it to
487  # estimate the location of the peak. It functions by finding the tap with
488  # the maximum value, and then fitting a quadratic to the points
489  # surrounding the maximum value tap. The number of points to which the
490  # quadratic is fitted is determined by the value of #peakFitPoints; the
491  # number of points is equal to \f$2 * n + 1\f$ where \f$n = \f$
492  # #peakFitPoints.
493  #
494  # The delay estimate that is returned is in units of #__dT__. So, a returned
495  # value of 2 would imply that the peak is located at 2, and therefore the
496  # delay corresponding to the correlation vector is 2 #__dT__.
497  #
498  # The returned delay may not include previously accumulated #signalDelay
499  # between the signal and the measurements. See the #storeStateVector
500  # function for more information on how the #signalDelay is stored and
501  # accumulated delay is accounted for.
502  #
503  # @param self The object pointer
504  # @param c The correlation vector
505  # @param P the correlation vector covariance matrix
506  #
507  # @return The estimate of the delay
508  def computeSignalDelay(
509  self,
510  c,
511  P
512  ):
513 
514  # First estimate of peak location is the location of the max value
515  peakLocation = np.argmax(c)
516 
517  # Next, we "roll" the correlation vector so that the values being
518  # fitted quadratically are the first 2 * peakFitPoints + 1 values
519  rollFactor = self.peakFitPoints - peakLocation
520 
521  if rollFactor != 0:
522  slicedC = np.roll(
523  c,
524  rollFactor
525  )
526  slicedP = (
527  np.roll(
528  np.roll(P,
529  rollFactor,
530  axis=0),
531  rollFactor,
532  axis=1
533  )
534  )
535  else:
536  slicedC = c
537  slicedP = P
538 
539  # Extract the portion of the correlation vector used for fitting
540  # quadratic
541  mySlice = slice(0, self.peakFitPoints * 2 + 1)
542 
543  slicedC = slicedC[mySlice]
544  slicedP = slicedP[mySlice, mySlice]
545 
546  # np.polyfit assumes that the weights will be the inverse standard
547  # deviation
548  weightVector = np.diag(slicedP)
549  weightVector = 1 / np.sqrt(weightVector)
550 
551  # xVec is the vector of "x" values corresponding the "y" values to
552  # which the quadratic is being fit.
553  xVec = np.linspace(0, self.peakFitPoints * 2, (self.peakFitPoints * 2) + 1)
554  xVec = xVec - rollFactor
555 
556  # Get the quadratic function that fits the peak and surrounding values,
557  # and use it to estimate the location of the max
558  quadraticVec = np.polyfit(xVec, slicedC, 2, w=weightVector)
559  delay = (-quadraticVec[1] / (2 * quadraticVec[0]))
560 
561  return delay

563  ## @fun #estimateSignalDelayUT uses a unscented tranform to estimate the
564  # delay corresponding to a correlation vector
565  #
566  # @details The #estimateSignalDelayUT method is responsible for computing
567  # the estimated value of delay corresponding to a correlation vector, as
568  # well as the variance of that estimate. These values are computed using
569  # a unscented transform (i.e. sigma-point) approach.
570  #
571  # The method receives the an estimate of the correlation vector, as well
572  # as the covariance matrix corresponding to that vector. From there it
573  # computes a set of n sigma points (where n is the length of the
574  # correlation vector), and for each of the generated sigma point vectors,
575  # it computes the peak location using the #computeSignalDelay method.
576  #
577  # @param self The object pointer
578  # @param h The correlation vector
579  # @param P The correlation vector covariance matrix
580  #
581  # @returns A dict containing the estimate of the peak location
582  # ("meanDelay") and the estimate variance ("varDelay")
583  def estimateSignalDelayUT(
584  self,
585  h,
586  P
587  ):
588  # Compute sigma points
589  hDimension = len(h)
590  sqrtP = np.linalg.cholesky(hDimension * P)
591 
592  sigmaPoints = h + np.append(sqrtP, -sqrtP, axis=0)
593 
594  sigmaPoints = np.append(np.array([h]), sigmaPoints, axis=0)
595 
596  # Initiate vector to store the resulting peaks from each sigma point
597  sigmaPointResults = np.zeros(len(sigmaPoints))
598 
599  # Compute the peak corresponding to each sigma point vector
600  for i in range(len(sigmaPoints)):
601  sigmaPointResults[i] = (
602  self.computeSignalOffset(sigmaPoints[i], P)
603  )
604 
605  meanDelay = np.mean(sigmaPointResults)
606  meanDelay = sigmaPointResults[0]
607  varDelay = np.var(sigmaPointResults)
608 
609  return {'meanDelay': meanDelay, 'varDelay': varDelay}
